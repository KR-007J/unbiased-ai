      } else {
        if (charIdx > 0) { setTypeText(target.slice(0, charIdx - 1)); setCharIdx(c => c - 1); }
        else { setDeleting(false); setTypeIdx(i => (i + 1) % TYPEWRITER_TEXTS.length); }
      }

    return () => clearTimeout(timer);

          ctx.globalAlpha = alpha * (1 - ping / 2) * 0.5;
          ctx.lineWidth = 1;
          ctx.stroke();
          ctx.globalAlpha = 1;
        }
